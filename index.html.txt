        // Fungsi Helper untuk Tombol Enter
        function addEnterListener(ids, action) {
            ids.forEach(id => {
                const el = document.getElementById(id);
                if (el) {
                    el.addEventListener('keydown', (e) => {
                        if (e.key === 'Enter') {
                            e.preventDefault();
                            action();
                        }
                    });
                }
            });
        }

        function checkAuthStatus() {
            const isAuth = sessionStorage.getItem('farm_auth_v482') === 'true';
            if (isAuth) {
                document.getElementById('login-screen').style.display = 'none';
                document.getElementById('main-app').style.display = 'block';
                initApp();
            } else {
                document.getElementById('login-screen').style.display = 'flex';
                document.getElementById('main-app').style.display = 'none';
                addEnterListener(['login-user', 'login-pass'], handleLogin);
            }
        }

        function handleLogin() {
            const u = document.getElementById('login-user').value;
            const p = document.getElementById('login-pass').value;
            const err = document.getElementById('login-error');
            const saved = localStorage.getItem('chickprofit_v482_data');
            if (saved) { const tmp = JSON.parse(saved); state.auth = tmp.auth; }

            if (u === state.auth.user && p === state.auth.pass) {
                sessionStorage.setItem('farm_auth_v482', 'true');
                err.classList.add('hidden');
                checkAuthStatus();
            } else { err.classList.remove('hidden'); }
        }

        function formatIDR(n) { return new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', maximumFractionDigits: 0 }).format(n); }

        function updateUI() {
            const totalJ = (state.stock.indukan_j || 0) + (state.stock.siapJual_j || 0);
            const totalB = (state.stock.indukan_b || 0) + (state.stock.siapJual_b || 0);
            const totalPop = totalJ + totalB + (state.stock.doc || 0);

            document.getElementById('grand-total-populasi').innerText = totalPop;
            document.getElementById('dash-jantan').innerText = totalJ;
            document.getElementById('dash-betina').innerText = totalB;
            document.getElementById('dash-doc').innerText = state.stock.doc;
            document.getElementById('total-telur-display').innerHTML = `${state.stock.telur} <span class="text-xs text-slate-400">Butir</span>`;
            document.getElementById('dash-pakan-dws').innerText = (state.stock.pakanDewasa || 0).toFixed(1) + " kg";
            document.getElementById('dash-pakan-doc').innerText = (state.stock.pakanDoc || 0).toFixed(1) + " kg";
            
            const today = new Date().toLocaleDateString('id-ID');
            const salesToday = state.history.sales.filter(s => s.date === today).reduce((acc, curr) => acc + curr.total, 0);
            document.getElementById('total-omzet-hari-ini').innerText = formatIDR(salesToday);

            document.getElementById('nav-farm-name').innerText = state.profile.farmName;
            
            const obatBox = document.getElementById('dash-obat-list');
            obatBox.innerHTML = '';
            if (state.inventory.obat.length === 0) {
                obatBox.innerHTML = '<span class="text-[7px] text-slate-300 italic">Kosong</span>';
            } else {
                state.inventory.obat.forEach(o => {
                    const s = document.createElement('span');
                    s.className = "bg-emerald-50 text-emerald-600 px-2 py-0.5 rounded-full text-[7px] font-black border border-emerald-100 mr-1 mb-1 inline-block";
                    s.innerText = `${o.name}: ${o.qty}`;
                    obatBox.appendChild(s);
                });
            }

            const logBox = document.getElementById('activity-log');
            logBox.innerHTML = '';
            state.history.logs.slice().reverse().forEach(log => {
                const d = document.createElement('div');
                d.className = "flex justify-between p-3 bg-slate-50 rounded-xl border-l-4 border-slate-300 mb-2";
                d.innerHTML = `<span><b>${log.time}</b> ${log.msg}</span> <span class="font-black opacity-30 uppercase">${log.type}</span>`;
                logBox.appendChild(d);
            });

            document.getElementById('up-ind-j').value = state.stock.indukan_j;
            document.getElementById('up-ind-b').value = state.stock.indukan_b;
            document.getElementById('up-sj-j').value = state.stock.siapJual_j;
            document.getElementById('up-sj-b').value = state.stock.siapJual_b;
            document.getElementById('up-doc').value = state.stock.doc;
        }

        function updateLogistik() {
            const pd = parseFloat(document.getElementById('stok-pakan-dws').value) || 0;
            const pc = parseFloat(document.getElementById('stok-pakan-doc').value) || 0;
            const oName = document.getElementById('nama-obat').value.trim();
            const oQty = parseInt(document.getElementById('qty-obat').value) || 0;
            if (pd > 0) state.stock.pakanDewasa += pd;
            if (pc > 0) state.stock.pakanDoc += pc;
            if (oName && oQty > 0) {
                const ex = state.inventory.obat.find(o => o.name.toLowerCase() === oName.toLowerCase());
                if (ex) ex.qty += oQty; else state.inventory.obat.push({ name: oName, qty: oQty });
            }
            addLog("LOGISTIK", "Update inventori"); saveState();
            ['stok-pakan-dws','stok-pakan-doc','nama-obat','qty-obat'].forEach(id => document.getElementById(id).value = '');
        }

        function saveDailyLog() {
            const t = parseInt(document.getElementById('input-telur').value) || 0;
            const p = parseFloat(document.getElementById('input-pakan-pakai').value) || 0;
            const m_in = parseInt(document.getElementById('input-mesin-in').value) || 0;
            const m_out = parseInt(document.getElementById('input-menetas').value) || 0;
            if (t > 0) state.stock.telur += t;
            if (p > 0) state.stock.pakanDewasa -= p;
            if (m_in > 0) state.stock.telur -= m_in;
            if (m_out > 0) state.stock.doc += m_out;
            addLog("HARIAN", "Simpan log harian"); saveState();
            ['input-telur','input-pakan-pakai','input-mesin-in','input-menetas'].forEach(id => document.getElementById(id).value = '');
        }

        function reportDeath() {
            const m = parseInt(document.getElementById('input-mati').value) || 0;
            const kat = document.getElementById('mati-kategori').value;
            if (m <= 0) return;
            if (kat === 'INDUKAN_J') state.stock.indukan_j -= m;
            else if (kat === 'INDUKAN_B') state.stock.indukan_b -= m;
            else if (kat === 'SIAP_JUAL_J') state.stock.siapJual_j -= m;
            else if (kat === 'SIAP_JUAL_B') state.stock.siapJual_b -= m;
            else if (kat === 'DOC') state.stock.doc -= m;
            addLog("MATI", `${m} ekor (${kat})`); saveState();
            document.getElementById('input-mati').value = '';
        }

        function updatePhysicalStock() {
            state.stock.indukan_j = parseInt(document.getElementById('up-ind-j').value) || 0;
            state.stock.indukan_b = parseInt(document.getElementById('up-ind-b').value) || 0;
            state.stock.siapJual_j = parseInt(document.getElementById('up-sj-j').value) || 0;
            state.stock.siapJual_b = parseInt(document.getElementById('up-sj-b').value) || 0;
            state.stock.doc = parseInt(document.getElementById('up-doc').value) || 0;
            addLog("OPNAME", "Koreksi stok fisik"); saveState(); alert("Stok Fisik Diperbarui!");
        }

        function calcTotal() {
            const q = parseInt(document.getElementById('jual-qty').value) || 0;
            const h = parseInt(document.getElementById('jual-harga').value) || 0;
            document.getElementById('calc-grand-total').innerText = formatIDR(q * h);
        }

        function processSale() {
            const cus = document.getElementById('jual-customer').value || "Umum";
            const qty = parseInt(document.getElementById('jual-qty').value) || 0;
            const hrg = parseInt(document.getElementById('jual-harga').value) || 0;
            const prd = document.getElementById('jual-produk').value;
            const jk = document.getElementById('jual-jk').value;
            if (qty <= 0 || hrg <= 0) return;
            if (prd === 'INDUKAN') { if (jk === 'J') state.stock.indukan_j -= qty; else state.stock.indukan_b -= qty; } 
            else if (prd === 'SIAP_JUAL') { if (jk === 'J') state.stock.siapJual_j -= qty; else state.stock.siapJual_b -= qty; } 
            else if (prd === 'DOC') state.stock.doc -= qty;
            else if (prd === 'TELUR') state.stock.telur -= qty;
            state.history.sales.push({ date: new Date().toLocaleDateString('id-ID'), customer: cus, total: qty * hrg });
            addLog("JUAL", `${qty} ${prd} terjual`); saveState();
            ['jual-qty','jual-harga','jual-customer','jual-kontak','jual-alamat'].forEach(id => document.getElementById(id).value = '');
        }

            saveState(); alert("Profil Disimpan!");
        }
